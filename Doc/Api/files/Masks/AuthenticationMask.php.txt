<?php
namespace kurapov\kurapov_validate\Masks;

use kurapov\kurapov_validate\Masks\AbstractMask;

class AuthenticationMask extends AbstractMask
{
    public function getMask()
    {
        return array(
            'email' =>array(
                'method' => 'email',
                'arguments' => 'null'
                'required' => true
            ),
            'password' => array(
                'method' => 'password',
                'arguments' => array(8, 30),
                'required' => true
            )
        );
    }
}
?>
